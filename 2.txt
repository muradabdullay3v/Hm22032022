using System;

class H2 {
    
  static void Main() {
    string Setir = "Hikmet Abbasov";
    NoSpace(ref Setir);
  }
  
  static void NoSpace(ref string rand){
      for(int i = 0 ; i < rand.Length ; i++){
          if(rand[i]==' '){
             continue;
          }
            Console.Write(rand[i]);
      }
  }
  
}
